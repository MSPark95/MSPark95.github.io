# 
화살표 함수
변수에 함수를 넣어주는것
~~~javascript
<script>
  let func=(a+b) =>{
    return a+b;
  }
</script>

// 한개의 인수인경우 생략이 가능하다
<script>
  let func= A =>{
    return;
  }
</script>
// 인수가 없더라도 괄호를 생략하면 안 된다.

// 연습
say("안녕하세요", let func=(a,b)=>a+b,say No)
~~~
